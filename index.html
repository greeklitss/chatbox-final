const socket = new WebSocket('wss://chatbox-final.onrender.com'); const chatWindow = document.getElementById('chat-window'); const messageInput = document.getElementById('message-input'); const usernameInput = document.getElementById('username'); const sendButton = document.getElementById('send-button'); // Λήψη Μηνυμάτων από τον Server socket.onmessage = function(event) { const message = event.data; const p = document.createElement('p'); p.textContent = message; chatWindow.appendChild(p); chatWindow.scrollTop = chatWindow.scrollHeight; }; // Αποστολή Μηνυμάτων στον Server sendButton.onclick = function() { const username = usernameInput.value || 'Ανώνυμος'; const message = messageInput.value; if (message) { // Στέλνουμε το μήνυμα και το όνομα const formattedMessage = `${username}: ${message}`; socket.send(formattedMessage); // Εμφανίζουμε το μήνυμα κατευθείαν const p = document.createElement('p'); p.textContent = formattedMessage; chatWindow.appendChild(p); chatWindow.scrollTop = chatWindow.scrollHeight; messageInput.value = ''; // Καθαρίζουμε το πεδίο } }; // Επίσης, στέλνουμε με το Enter messageInput.addEventListener('keydown', function(event) { if (event.key === 'Enter') { sendButton.click(); } });
